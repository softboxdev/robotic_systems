def calculate_bmi():
    """
    –ó–∞–¥–∞—á–∞: –†–∞—Å—Å—á–∏—Ç–∞—Ç—å –∏–Ω–¥–µ–∫—Å –º–∞—Å—Å—ã —Ç–µ–ª–∞ (BMI) —Å –ø—Ä–æ–≤–µ—Ä–∫–æ–π —Ç–∏–ø–æ–≤ –¥–∞–Ω–Ω—ã—Ö
    """
    try:
        # –í–≤–æ–¥ –¥–∞–Ω–Ω—ã—Ö —Å –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ–º —Ç–∏–ø–æ–≤
        weight = float(input("–í–≤–µ–¥–∏—Ç–µ –≤–µ—Å –≤ –∫–≥: "))    # float –¥–ª—è —Ç–æ—á–Ω–æ—Å—Ç–∏
        height = float(input("–í–≤–µ–¥–∏—Ç–µ —Ä–æ—Å—Ç –≤ –º–µ—Ç—Ä–∞—Ö: "))
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ –≤–∞–ª–∏–¥–Ω–æ—Å—Ç–∏ –¥–∞–Ω–Ω—ã—Ö
        if weight <= 0 or height <= 0:
            print("–û—à–∏–±–∫–∞: –≤–µ—Å –∏ —Ä–æ—Å—Ç –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–º–∏ —á–∏—Å–ª–∞–º–∏")
            return
        
        # –†–∞—Å—á–µ—Ç BMI
        bmi = weight / (height ** 2)  # –§–æ—Ä–º—É–ª–∞: –≤–µ—Å / —Ä–æ—Å—Ç¬≤
        
        # –õ–æ–≥–∏—á–µ—Å–∫–∏–µ –ø—Ä–æ–≤–µ—Ä–∫–∏
        is_underweight = bmi < 18.5
        is_normal = 18.5 <= bmi < 25
        is_overweight = 25 <= bmi < 30
        is_obese = bmi >= 30
        
        print(f"\n–í–∞—à BMI: {bmi:.2f}")  # :.2f - –æ–∫—Ä—É–≥–ª–µ–Ω–∏–µ –¥–æ 2 –∑–Ω–∞–∫–æ–≤ –ø–æ—Å–ª–µ –∑–∞–ø—è—Ç–æ–π
        print("–î–∏–∞–≥–Ω–æ–∑:")
        print(f"–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω—ã–π –≤–µ—Å: {is_underweight}")
        print(f"–ù–æ—Ä–º–∞–ª—å–Ω—ã–π –≤–µ—Å: {is_normal}")
        print(f"–ò–∑–±—ã—Ç–æ—á–Ω—ã–π –≤–µ—Å: {is_overweight}")
        print(f"–û–∂–∏—Ä–µ–Ω–∏–µ: {is_obese}")
        
    except ValueError:
        print("–û—à–∏–±–∫–∞: –≤–≤–µ–¥–∏—Ç–µ —á–∏—Å–ª–æ–≤—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è!")

# calculate_bmi()
```

### –ó–∞–¥–∞—á–∞ 2: –ö–æ–Ω–≤–µ—Ä—Ç–µ—Ä –≤–∞–ª—é—Ç

def currency_converter():
    """
    –ó–∞–¥–∞—á–∞: –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä–æ–≤–∞—Ç—å –≤–∞–ª—é—Ç—É —Å –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ–º —Ä–∞–∑–Ω—ã—Ö —á–∏—Å–ª–æ–≤—ã—Ö –æ–ø–µ—Ä–∞—Ü–∏–π
    """
    # –ö—É—Ä—Å—ã –≤–∞–ª—é—Ç (float –¥–ª—è —Ç–æ—á–Ω–æ—Å—Ç–∏)
    usd_to_rub = 91.50    # 1 USD = 91.50 RUB
    eur_to_rub = 98.25    # 1 EUR = 98.25 RUB
    
    print("–ö–æ–Ω–≤–µ—Ä—Ç–µ—Ä –≤–∞–ª—é—Ç")
    print("1. USD to RUB")
    print("2. EUR to RUB")
    print("3. RUB to USD")
    print("4. RUB to EUR")
    
    choice = input("–í—ã–±–µ—Ä–∏—Ç–µ –æ–ø–µ—Ä–∞—Ü–∏—é (1-4): ")
    amount = float(input("–í–≤–µ–¥–∏—Ç–µ —Å—É–º–º—É: "))
    
    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–æ–µ —á–∏—Å–ª–æ
    if amount <= 0:
        print("–°—É–º–º–∞ –¥–æ–ª–∂–Ω–∞ –±—ã—Ç—å –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–æ–π!")
        return
    
    # –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –∫–æ–Ω–≤–µ—Ä—Ç–∞—Ü–∏–∏
    if choice == "1":
        result = amount * usd_to_rub
        print(f"{amount} USD = {result:.2f} RUB")  # –û–∫—Ä—É–≥–ª–µ–Ω–∏–µ –¥–æ –∫–æ–ø–µ–µ–∫
    elif choice == "2":
        result = amount * eur_to_rub
        print(f"{amount} EUR = {result:.2f} RUB")
    elif choice == "3":
        result = amount / usd_to_rub
        print(f"{amount} RUB = {result:.2f} USD")
    elif choice == "4":
        result = amount / eur_to_rub
        print(f"{amount} RUB = {result:.2f} EUR")
    else:
        print("–ù–µ–≤–µ—Ä–Ω—ã–π –≤—ã–±–æ—Ä –æ–ø–µ—Ä–∞—Ü–∏–∏!")

# currency_converter()
```

### –ó–∞–¥–∞—á–∞ 3: –ü—Ä–æ–≤–µ—Ä–∫–∞ –ø—Ä–æ—Å—Ç–æ–≥–æ —á–∏—Å–ª–∞

def is_prime_number():
    """
    –ó–∞–¥–∞—á–∞: –ü—Ä–æ–≤–µ—Ä–∏—Ç—å, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —á–∏—Å–ª–æ –ø—Ä–æ—Å—Ç—ã–º
    """
    try:
        number = int(input("–í–≤–µ–¥–∏—Ç–µ —Ü–µ–ª–æ–µ —á–∏—Å–ª–æ: "))
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ –≥—Ä–∞–Ω–∏—á–Ω—ã—Ö —Å–ª—É—á–∞–µ–≤
        if number < 2:
            print(f"{number} –Ω–µ —è–≤–ª—è–µ—Ç—Å—è –ø—Ä–æ—Å—Ç—ã–º —á–∏—Å–ª–æ–º")
            return
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø—Ä–æ—Å—Ç–æ—Ç—É
        is_prime = True  # –ë—É–ª–µ–≤–∞ –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è-—Ñ–ª–∞–≥
        
        for i in range(2, int(number ** 0.5) + 1):  # –ü—Ä–æ–≤–µ—Ä—è–µ–º –¥–æ –∫–æ—Ä–Ω—è –∏–∑ —á–∏—Å–ª–∞
            if number % i == 0:  # –ï—Å–ª–∏ –¥–µ–ª–∏—Ç—Å—è –±–µ–∑ –æ—Å—Ç–∞—Ç–∫–∞
